CREATE OR REPLACE VIEW dept_grades AS
SELECT id, name, dept_name, ROUND(SUM(point * credits) / SUM(credits), 1) AS GPA
FROM student 
NATURAL LEFT OUTER JOIN (
    SELECT id, takes.course_id, course.credits, MAX(point) AS point
    FROM takes, gp_mapping, course
    WHERE TRIM(takes.grade) = gp_mapping.grade 
      AND takes.course_id = course.course_id
    GROUP BY id, takes.course_id, course.credits
) AS tmp
GROUP BY id, name;
